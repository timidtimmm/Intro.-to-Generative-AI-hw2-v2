              ["tools/call","Invoke a tool","name, arguments"],

              ["logs/list","Get call log entries","limit?"],
              ["logs/clear","Clear all logs","—"],

Example: tools/call

          
// Request
{
  "jsonrpc": "2.0",
  "id": 1,
  "method": "tools/call",
  "params": {
    "name": "calculator",
    "arguments": { "expression": "2 * (3 + 4)" }
  }
}

// Response
{
  "jsonrpc": "2.0",
  "id": 1,
  "result": {
    "content": [{"type": "text", "text": "{\"result\": 14}"}],
    "isError": false,
    "_latency_ms": 0.4
  }
}
